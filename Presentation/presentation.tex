% ------------------------------------------
% GeMFreq and AttHead Details
% ------------------------------------------
\begin{frame}[t]{Key Architecture Components}
    \begin{columns}[T]
        \begin{column}{0.48\textwidth}
            \begin{block}{GeMFreq Pooling}
                \textbf{Generalized Mean Frequency Pooling}
                \begin{itemize}
                    \item Learnable pooling parameter $p$
                    \item Formula: $\left(\frac{1}{|F|}\sum_{f} x_f^p\right)^{1/p}$
                    \item Adapts between max and average pooling
                    \item Better captures frequency patterns
                \end{itemize}
            \end{block}

            \begin{exampleblock}{Why Features Only?}
                Extract spatial features without classification head $\rightarrow$ preserves temporal dimension for SED
            \end{exampleblock}
        \end{column}
        \begin{column}{0.48\textwidth}
            \begin{block}{AttHead (Attention Head)}
                \begin{enumerate}
                    \item Dense layer: backbone\_dim $\rightarrow$ 512
                    \item Parallel branches:
                    \begin{itemize}
                        \item \textbf{Attention}: 1$\times$1 Conv $\rightarrow$ Softmax
                        \item \textbf{Logits}: 1$\times$1 Conv $\rightarrow$ 206 classes
                    \end{itemize}
                    \item Weighted aggregation:
                    $$\text{clip\_logits} = \sum_t \text{att}_t \cdot \text{logits}_t$$
                \end{enumerate}
            \end{block}

            \textbf{Advantage}: Learns which time frames contain species calls!
        \end{column}
    \end{columns}
\end{frame}

                    \end{tikzpicture}
                \end{center}
                $\tilde{x} = \lambda x_a + (1-\lambda) x_b$

                \textcolor{birdgreen}{\textbf{+2-3\% accuracy}}
            \end{block}
        \end{column}
        \begin{column}{0.32\textwidth}
            \begin{block}{SpecAugment}
                \begin{center}
                    \begin{tikzpicture}[scale=0.55]
                        \draw[fill=skyblue!30] (0,0) rectangle (3,2);
                        \draw[fill=black] (1,0) rectangle (1.3,2);
                        \draw[fill=black] (0,0.8) rectangle (3,1.1);
                        \node at (1.5,-0.3) {\tiny Masked};
                    \end{tikzpicture}
                \end{center}
                \begin{itemize}
                    \item Time masking: 20\%
                    \item Freq masking: 10\%
                \end{itemize}
                \textcolor{birdgreen}{\textbf{+1-2\% accuracy}}
            \end{block}
        \end{column}
        \begin{column}{0.32\textwidth}
            \begin{block}{Weighted Sampling}
                Address class imbalance:
                $$w_c = \frac{1}{\sqrt{n_c}}$$
                \begin{itemize}
                    \item Rare species $\uparrow$
                    \item Common species $\downarrow$
                \end{itemize}
                \textcolor{birdgreen}{\textbf{Critical for rare species}}
            \end{block}
        \end{column}
    \end{columns}
\end{frame}

% ------------------------------------------
% Training Configuration
% ------------------------------------------
\begin{frame}[t]{Training Configuration}
    \begin{columns}[T]
        \begin{column}{0.48\textwidth}
            \begin{block}{Optimizer \& Scheduler}
                \begin{itemize}
                    \item \textbf{Optimizer}: AdamW
                    \begin{itemize}
                        \item Learning rate: $5 \times 10^{-4}$
                        \item Weight decay: $10^{-4}$
                    \end{itemize}
                    \item \textbf{Scheduler}: CosineAnnealingWarmRestarts
                    \begin{itemize}
                        \item $T_0 = 5$ epochs
                        \item $\eta_{min} = 10^{-6}$
                    \end{itemize}
                    \item \textbf{Gradient Clipping}: max\_norm = 1.0
                \end{itemize}
            \end{block}
        \end{column}
        \begin{column}{0.48\textwidth}
            \begin{block}{Loss Function}
                \textbf{CrossEntropy with Label Smoothing}
                $$\mathcal{L} = -\sum_c y_c' \log(p_c)$$
                where $y_c' = (1-\epsilon)y_c + \frac{\epsilon}{C}$

                \begin{itemize}
                    \item Label smoothing: $\epsilon = 0.1$
                    \item Prevents overconfident predictions
                    \item Better calibration
                \end{itemize}
            \end{block}

            \begin{exampleblock}{Training Details}
                \begin{itemize}
                    \item Epochs: 15-35
                    \item Batch size: 64
                    \item 5-fold cross-validation
                \end{itemize}
            \end{exampleblock}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}{Overlap-Average Inference}
    \begin{center}
        \begin{tikzpicture}[scale=0.85]
            % 60s soundscape
            \draw[fill=skyblue!20, rounded corners] (0,3) rectangle (12,3.8);
            \node at (6,3.4) {60-second Soundscape};

            % Segments
            \foreach \i in {0,1,2,3} {
                \pgfmathsetmacro{\start}{\i*2}
                \pgfmathsetmacro{\midy}{2 - \i*0.5}
                \draw[fill=green!\i 0!blue, opacity=0.5, rounded corners]
                    (\start,\midy) rectangle (\start+4,\midy+0.4);
                \node[font=\tiny] at (\start+2,\midy+0.2) {Chunk \pgfmathparse{int(\i+1)}\pgfmathresult};
            }

            % Overlap indication
            \draw[<->, thick, red] (2,1.2) -- (4,1.2);
            \node[font=\tiny, red] at (3,0.9) {Overlap};

            % Arrow to averaging
            \draw[->, thick] (6,0.5) -- (6,-0.2);

            % Averaged predictions
            \draw[fill=purple!30, rounded corners] (2,-0.5) rectangle (10,-1.1);
            \node[font=\small] at (6,-0.8) {Overlap-Averaged Framewise Predictions};

            % Final output
            \draw[->, thick] (6,-1.4) -- (6,-1.8);
            \draw[fill=orange!30, rounded corners] (3,-2) rectangle (9,-2.6);
            \node[font=\small] at (6,-2.3) {12 $\times$ 206 Prediction Matrix};
        \end{tikzpicture}
    \end{center}

    \vspace{0.2cm}
    \begin{columns}[T]
        \begin{column}{0.48\textwidth}
            \textbf{Process:}
            \begin{enumerate}
                \item Split into 12 overlapping 20s chunks (5s step)
                \item Each chunk $\rightarrow$ framewise predictions
                \item Average overlapping frame predictions
            \end{enumerate}
        \end{column}
        \begin{column}{0.48\textwidth}
            \textbf{Benefits:}
            \begin{itemize}
                \item More robust predictions
                \item Implicit TTA effect
                \item +0.2-0.3\% accuracy boost
            \end{itemize}
        \end{column}
    \end{columns}
\end{frame}

% ------------------------------------------
% Delta-Shift TTA and Smoothing
% ------------------------------------------
\begin{frame}[t]{Delta-Shift TTA \& Gaussian Smoothing}
    \begin{columns}[T]
        \begin{column}{0.48\textwidth}
            \begin{block}{Delta-Shift TTA}
                Temporal shift augmentation at inference:
                \begin{center}
                    \begin{tikzpicture}[scale=0.7]
                        \draw[fill=blue!20] (0,2) rectangle (4,2.5);
                        \node at (2,2.25) {\tiny Original};

                        \draw[fill=blue!30] (-0.3,1) rectangle (3.7,1.5);
                        \node at (1.7,1.25) {\tiny Shift -2};

                        \draw[fill=blue!30] (0.3,0) rectangle (4.3,0.5);
                        \node at (2.3,0.25) {\tiny Shift +2};

                        \draw[->, thick] (4.5,1.25) -- (5.5,1.25);

                        \draw[fill=green!30] (5.7,0.8) rectangle (7.7,1.7);
                        \node at (6.7,1.25) {\tiny Blend};
                    \end{tikzpicture}
                \end{center}

                \textbf{Blending weights:}
                $$\text{pred} = 0.50 \cdot p_0 + 0.25 \cdot p_{-2} + 0.25 \cdot p_{+2}$$
            \end{block}
        \end{column}
        \begin{column}{0.48\textwidth}
            \begin{block}{Gaussian Smoothing}
                1D convolution to reduce prediction jitter:

                \textbf{Kernel:} $[0.1, 0.2, 0.4, 0.2, 0.1]$

                \begin{center}
                    \begin{tikzpicture}[scale=0.6]
                        % Before
                        \draw[thick, blue] plot[smooth] coordinates {
                            (0,1) (0.5,2) (1,1.2) (1.5,2.5) (2,1.8) (2.5,2.2) (3,1.5)
                        };
                        \node at (1.5,-0.3) {\tiny Before};

                        % Arrow
                        \draw[->, thick] (3.5,1.5) -- (4.5,1.5);

                        % After
                        \draw[thick, green!60!black] plot[smooth] coordinates {
                            (5,1.3) (5.5,1.7) (6,1.8) (6.5,2) (7,1.9) (7.5,1.8) (8,1.6)
                        };
                        \node at (6.5,-0.3) {\tiny After};
                    \end{tikzpicture}
                \end{center}

                Reduces temporal noise while preserving patterns.
            \end{block}
        \end{column}
    \end{columns}

    \vspace{0.2cm}
    \begin{exampleblock}{Combined Effect}
        Delta-Shift TTA + Gaussian Smoothing = \textbf{Free accuracy boost at inference time!}
    \end{exampleblock}
\end{frame}
