% ------------------------------------------
% GeMFreq and AttHead Details
% ------------------------------------------
\begin{frame}[t]{Key Architecture Components}
    \begin{columns}[T, onlytextwidth]
        \begin{column}{0.48\textwidth}
            \begin{tcolorbox}[draculabox={GeMFreq Pooling}]
                \scriptsize
                \textbf{Generalized Mean Freq Pooling}
                \begin{itemize}
                    \item Formula: $\left(\frac{1}{|F|}\sum_{f} x_f^p\right)^{1/p}$
                    \item \textbf{Learnable} parameter $p$
                    \item Adapts: Max vs Average pooling
                    \item Captures fine-grained freq patterns
                \end{itemize}
            \end{tcolorbox}

            \begin{tcolorbox}[greenbox_tcb={Features Only}]
                \scriptsize Preserves temporal dimension by removing classification head from backbone.
            \end{tcolorbox}
        \end{column}
        \begin{column}{0.48\textwidth}
            \begin{tcolorbox}[draculabox={AttHead (Attention)}]
                \scriptsize
                \begin{enumerate}
                    \item \textbf{Attention}: 1$\times$1 Conv $\rightarrow$ Softmax
                    \item \textbf{Logits}: 1$\times$1 Conv $\rightarrow$ 206
                    \item \textbf{Aggregation}:
                    $$\text{clip\_logits} = \sum_t \text{att}_t \cdot \text{logits}_t$$
                \end{enumerate}
                \textcolor{draculagreen}{\textbf{Benefit}}: Learns \textit{when} species call!
            \end{tcolorbox}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}[t]{Training Configuration}
    \begin{columns}[T, onlytextwidth]
        \begin{column}{0.48\textwidth}
            \begin{tcolorbox}[draculabox={Optimizer \& Scheduler}]
                \scriptsize
                \begin{itemize}
                    \item \textbf{Optimizer}: AdamW
                    \begin{itemize}
                        \item LR: $5 \times 10^{-4}$ $\cdot$ WD: $10^{-4}$
                    \end{itemize}
                    \item \textbf{Scheduler}: CosineAnnealing
                    \begin{itemize}
                        \item $T_0 = 5$ epochs
                        \item $\eta_{min} = 10^{-6}$
                    \end{itemize}
                    \item \textbf{Grad Clip}: max\_norm = 1.0
                \end{itemize}
            \end{tcolorbox}
        \end{column}
        \begin{column}{0.48\textwidth}
            \begin{tcolorbox}[orangebox_tcb={Loss Function}]
                \scriptsize
                \textbf{CE + Label Smoothing}
                $$\mathcal{L} = -\sum_c y_c' \log(p_c)$$
                \begin{itemize}
                    \item \textbf{Smoothing}: $\epsilon = 0.1$
                    \item Prevents overconfidence
                \end{itemize}
            \end{tcolorbox}

            \begin{tcolorbox}[greenbox_tcb={Setup}]
                \scriptsize 15-35 Epochs $\cdot$ Batch 64 $\cdot$ 5-Fold CV
            \end{tcolorbox}
        \end{column}
    \end{columns}
\end{frame}
